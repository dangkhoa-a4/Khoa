-- [5] Hàm Tween (Bay xuyên đảo)
function Tween(pos)
    local char = player.Character or player.CharacterAdded:Wait()
    local root = char:WaitForChild("HumanoidRootPart")
    local dist = (root.Position - pos.p).Magnitude
    local speed = 320 -- Tốc độ chuẩn Salio
    local info = TweenInfo.new(dist/speed, Enum.EasingStyle.Linear)
    local twin = TweenService:Create(root, info, {CFrame = pos})
    twin:Play()
    return twin
end

-- [12] Hàm lấy thông tin nhiệm vụ phù hợp với Level hiện tại
function GetCurrentQuest()
    local myLvl = player.Data.Level.Value
    local target = MobData[1]
    for _, v in pairs(MobData) do
        if myLvl >= v.Level then
            target = v
        end
    end
    return target
end

-- [13] Hàm kiểm tra quái có tồn tại trong Workspace không
function GetTargetMob(mobName)
    for _, v in pairs(game.Workspace.Enemies:GetChildren()) do
        if v.Name == mobName and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
            return v
        end
    end
    return nil
end

-- [14] Logic tự động farm (Kaitun Loop)
spawn(function()
    while _G.Kaitun do
        task.wait(0.5)
        pcall(function()
            local questInfo = GetCurrentQuest()
            local hasQuest = player.Data.CurrentQuest.Value -- Check folder quest của bạn
            
            if hasQuest == "" then
                -- Nếu chưa có quest thì bay đi nhận quest
                status.Text = "Status: Đang đi nhận nhiệm vụ " .. questInfo.MobName
                local npc = game.Workspace.NPCs:FindFirstChild(questInfo.NPC)
                if npc then
                    Tween(npc.HumanoidRootPart.CFrame)
                    -- Remote nhận quest (Sửa lại tên theo RemoteSpy của bạn)
                    ReplicatedStorage.Remotes.QuestEvent:FireServer("Claim", questInfo.QuestName)
                end
            else
                -- Nếu có quest rồi thì đi tìm quái
                local target = GetTargetMob(questInfo.MobName)
                if target then
                    status.Text = "Status: Đang farm " .. questInfo.MobName
                    -- Sử dụng hàm bay trên đầu quái đã viết ở Phần 1
                    Tween(GetFarmPos(target))
                else
                    -- Nếu không có quái thì bay về bãi spawn quái đợi
                    status.Text = "Status: Đợi quái xuất hiện..."
                    Tween(game.Workspace.Spawns[questInfo.MobName].CFrame)
                end
            end
        end)
    end
end)

-- [32] Hàm kiểm tra nguyên liệu trong kho (Inventory Check)
function HasMaterials(itemList)
    if not itemList then return true end
    local inv = player.Data.Inventory
    for i = 1, #itemList, 2 do
        local name = itemList[i]
        local amount = itemList[i+1]
        if not inv:FindFirstChild(name) or inv[name].Value < amount then
            return false
        end
    end
    return true
end
